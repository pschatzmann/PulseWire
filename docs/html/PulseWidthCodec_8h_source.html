68 bool decodeByte(Vector<OutputEdge>& edges, uint8_t& result) override {
69 if (edges.size() < 16) {

73 uint8_t byte = 0;
74 int bit = 0;

79 byte |= (1 << (7 - bit));
80 } else if (bitMatch(edge.pulseUs, false)) {
81 // bit is 0
82 } else {

84 edge.pulseUs);
85 }
86 bit++;
87 }
