53 size_t encodeBit(bool bit, Vector<OutputEdge>& output) override {
54 OutputEdge pulse, space;
55 if (bit) {
56 // Logical '1': long pulse, short space
57 pulse.level = true; // HIGH
58 pulse.pulseUs = _longPulseUs;
59 space.level = false; // LOW

88 bool decodeByte(Vector<OutputEdge>& edges, uint8_t& result) const override {
89 assert(edges.size() == getBitCount());
90 bool valid = true;
91 uint8_t& byte = result;
92 for (int i = 0; i < 8; ++i) {
93 // For pulse-distance, each bit is represented by a pulse (HIGH) followed
94 // by a space (LOW) We check the duration of the pulse to determine the
95 // bit value
96 if (bitMatch(edges[i * 2].pulseUs, true)) {
97 byte |= (1 << (7 - i)); // Logical '1'
98 } else if (bitMatch(edges[i * 2].pulseUs, false)) {
99 // Logical '0', do nothing
100 } else {
101 // Invalid pulse, return 0 or set a valid flag if needed
102 valid = false;
103 }
104 }
105 return valid;
106 }

108 bool bitMatch(uint32_t duration, bool bit) const {
109 uint32_t expected = bit ? _longPulseUs : _shortPulseUs;
110 return (duration >= expected - _toleranceUs &&
111 duration <= expected + _toleranceUs);
