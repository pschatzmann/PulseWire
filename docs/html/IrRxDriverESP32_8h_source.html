118 size_t readBytes(uint8_t* buffer, size_t length) override {
119 size_t totalRead = 0;
120 // First, serve from rxOverflow if any
121 while (length > 0 && rxOverflow.available() > 0) {
122 int b = rxOverflow.read();
123 if (b < 0) break;
124 *buffer++ = (uint8_t)b;
125 --length;
126 ++totalRead;
127 }
128 // Then, read full frames from the queue as needed
129 uint8_t frame[_frameSize]{};
130 while (length > 0 && _frameQueue &&
131 uxQueueMessagesWaiting(_frameQueue) > 0) {
132 if (xQueueReceive(_frameQueue, frame, 0) == pdTRUE) {
133 size_t toCopy = (length < _frameSize) ? length : _frameSize;
134 memcpy(buffer, frame, toCopy);
135 totalRead += toCopy;
136 buffer += toCopy;
137 length -= toCopy;
138 // If we didn't consume the whole frame, store the rest in rxOverflow
139 if (toCopy < _frameSize) {
140 rxOverflow.writeArray(frame + toCopy, _frameSize - toCopy);
141 }
142 }
143 }
144 return totalRead;
145 }

180 static void rxTask(void* arg) {
181 auto* self = static_cast<RxDriverESP32*>(arg);
182 size_t symbolCount = self->_frameSize * self->_codec.getBitCount();
183 Vector<rmt_symbol_word_t> symbols(symbolCount);
184 uint8_t data[self->_frameSize]{};
185 self->_lastSymbolTime = micros();
186 while (!self->_stopTask) {
187 size_t rx_size = 0;
188 rmt_receive_config_t rx_cfg = {};
189 esp_err_t err =
190 rmt_receive(self->_rxChannel, symbols.data(),
191 symbols.size() * sizeof(rmt_symbol_word_t), &rx_cfg);
192 bool gotSymbol = false;
193 if (err == ESP_OK && rx_size > 0) {
194 size_t num_bits = rx_size / sizeof(rmt_symbol_word_t);
195 // Use decodeEdge for each symbol
196 uint32_t bitPeriodUs = 1000000UL / self->_freqHz;
197 uint32_t minUs = bitPeriodUs / 2;
198 uint32_t maxUs = bitPeriodUs * 2;
199 size_t frameLen = 0;
200 for (size_t i = 0; i < num_bits; ++i) {
201 uint32_t duration = symbols[i].duration0 + symbols[i].duration1;
202 bool isOne = (symbols[i].level0 == 1 && symbols[i].level1 == 0);
203 bool isZero = (symbols[i].level0 == 0 && symbols[i].level1 == 1);
204 if (isOne || isZero) {
205 bool level = isOne ? 1 : 0;
206 self->_lastSymbolTime = micros();
207 gotSymbol = true;
208 if (self->_codec.decodeEdge(duration, level, minUs, maxUs, data,
209 self->_frameSize, frameLen)) {
210 if (self->_useChecksum) {
211 // Validate checksum
212 if (frameLen < 2) continue;
213 uint8_t sum = 0;
214 for (size_t j = 0; j < frameLen - 1; ++j) sum += data[j];
215 if (sum == data[frameLen - 1]) {
216 xQueueSend(self->_frameQueue, data, 0);
217 }
218 } else {
219 xQueueSend(self->_frameQueue, data, 0);
220 }
221 }
222 }
223 }
224 }
225 // Timeout logic: if no symbol received for timeoutUs, flush buffer
226 uint32_t now = micros();
227 if (!gotSymbol && (now - self->_lastSymbolTime > self->_timeoutUs)) {
228 // Try to decode with a zero-duration edge to flush the buffer
229 size_t frameLen = 0;
230 if (self->_codec.decodeEdge(0, 0, 0, 0, data, self->_frameSize, frameLen)) {
231 if (self->_useChecksum) {
232 if (frameLen >= 2) {
233 uint8_t sum = 0;
234 for (size_t j = 0; j < frameLen - 1; ++j) sum += data[j];
235 if (sum == data[frameLen - 1]) {
236 xQueueSend(self->_frameQueue, data, 0);
237 }
238 }
239 } else {
240 xQueueSend(self->_frameQueue, data, 0);
241 }
242 }
243 self->_lastSymbolTime = now;
244 }
245 delay(1);
246 }
247 vTaskDelete(nullptr);
248 }
