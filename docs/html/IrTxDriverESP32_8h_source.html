155 void flush() {
156 output.clear();
157 // Insert preamble edges at the start
158 _codec.getPreambleDetector().getEdges(output);
159 int n = _txBuffer.available();
160 if (n == 0) return;
161 // process bytes
162 uint8_t sum = 0;
163 for (int byteIdx = 0; byteIdx < n; ++byteIdx) {
164 int val = _txBuffer.read();
165 uint8_t b = (val >= 0) ? (uint8_t)val : 0;
166 sum += b;
167 _codec.encode(b, _bitPeriodUs, output);
168 }
169 // Optionally append checksum
170 if (_useChecksum) {
171 _codec.encode(sum, _bitPeriodUs, output);
172 }
173 // Convert OutputSpec to RMT symbols (step by 2)
174 size_t symbolCount = output.size() / 2;
175 if (_itemsBuffer.size() < symbolCount) _itemsBuffer.resize(symbolCount);
176 for (size_t i = 0, j = 0; i + 1 < output.size(); i += 2, ++j) {
177 _itemsBuffer[j].duration0 = output[i].pulseUs;
178 _itemsBuffer[j].level0 = output[i].level ? 1 : 0;
179 _itemsBuffer[j].duration1 = output[i + 1].pulseUs;
180 _itemsBuffer[j].level1 = output[i + 1].level ? 1 : 0;
181 }
182 if (_txChannel && _txEncoder) {
183 rmt_transmit(_txChannel, _txEncoder, _itemsBuffer.data(),
184 symbolCount * sizeof(rmt_symbol_word_t), &_txTransmitConfig);
185 }
186 }
