41 public:
42 virtual ~Preamble() = default;
43
44 virtual bool begin(uint32_t bitFrequencyHz) {
45 _history.reserve(preambleLength());
46 return true;
47 }
48

52 virtual bool detect(const OutputEdge& edge) {
53 Logger::debug("Preamble Detecting edge: level=%s, pulseUs=%d", edge.level ? "HIGH" : "LOW",
54 edge.pulseUs);
55 size_t N = preambleLength();
56 if (N == 0) return true;
57 _history.push_back(edge);
58 if (_history.size() > N) _history.erase(_history.begin());
59 if (_history.size() < N) {
60 Logger::debug("Not enough edges for preamble detection: %d/%d",
61 _history.size(), N);
62 return false;
63 }
64
65 for (size_t i = 0; i < N; ++i) {
66 if (_history[i].level != _expected[i].level) {
67 Logger::debug("Invalid level idx %d: expected %s, got %s - %d us", i,
68 _expected[i].level ? "HIGH" : "LOW",
69 _history[i].level ? "HIGH" : "LOW ",
70 _history[i].pulseUs);
71 return false;
72 }
73 // Use 20% tolerance or 50us minimum
74 uint32_t tol = 0.2 * _expected[i].pulseUs;
75 if (tol < 50) tol = 50;
76 if (!inRange(_history[i].pulseUs, _expected[i].pulseUs, tol)) {
77 Logger::debug("Invalid pulse duration: expected %d us, got %d us",
78 _expected[i].pulseUs, _history[i].pulseUs);
79 return false;
80 }
81 }
82 Logger::debug("Preamble detected");
83 _history.clear();
84 return true;
85 }
86
88 virtual int getEdges(
89 pulsewire::Vector<pulsewire::OutputEdge>& output) const = 0;
90
91 virtual size_t preambleLength() const = 0;
92
95 bool inRange(uint32_t value, uint32_t target, uint32_t tolerance) const {
96 return (value >= target - tolerance) && (value <= target + tolerance);
97 }

99 protected:

202class ManchesterPreamble : public CustomPreambleUs {
203 public:
204 ManchesterPreamble(uint8_t runInCycles = 2) : _runInCycles(runInCycles) {}
205
206 bool begin(uint32_t bitFrequencyHz) override {
207 constexpr uint8_t MAX_RUN_IN_CYCLES = 32;
208 if (_runInCycles == 0) {
209 _runInCycles = 2;
210 } else if (_runInCycles > MAX_RUN_IN_CYCLES) {
211 Logger::error("ManchesterPreamble: runInCycles=%d too large, clamping to %d",
212 _runInCycles, MAX_RUN_IN_CYCLES);
213 _runInCycles = MAX_RUN_IN_CYCLES;
214 }
215
216 // double the frequency for Manchester since each bit is two edges
217 _history.clear();
218 size_t runInEdges = static_cast<size_t>(_runInCycles) * 2;
219 _history.reserve(runInEdges + 1); // run-in edges + start pulse
220 _expected.reserve(runInEdges + 1);
221 _freqHz = bitFrequencyHz * 2;
222 _bitPeriod = 1000000 / _freqHz;
223
224 // Generate the expected edges for the Manchester preamble
225 _expected.clear();
226 bool level = true;
227 // Run-in: alternating edges for clock recovery
228 for (size_t i = 0; i < runInEdges; ++i) {
229 addEdge(level, _bitPeriod);
230 level = !level;
231 }
232
233 return true;
234 }
235
236 private:
237 uint8_t _runInCycles = 0;
238 uint32_t _freqHz = 0;
239 uint32_t _bitPeriod = 0;
240};
